﻿goog.provide('KMap.WMTSLayer');

goog.require('KMap');
goog.require('KMap.Layer');

/**
 * @api
 * @constructor
 * @extends {KMap.Layer}
 * @param {string} id id.
 * @param {Object} options options.
 */
KMap.BaiduLayer = function (id, options) {
    var baseOptions = ol.obj.assign({}, options);
    KMap.Layer.call(this, id, baseOptions);

    this.transform = new KMap.Transform();
};
ol.inherits(KMap.BaiduLayer, KMap.Layer);

KMap.BaiduLayer.prototype.createLayer = function (options) {
    var wmts_layer = new ol.layer.Tile();

    var wmts_options = ol.source.WMTS.optionsFromCapabilities(result, options);
    wmts_options.layer = wmts_options.layer || '';
    wmts_options.matrixSet = wmts_options.matrixSet || 'EPSG:4326';

    var wmts_source = new ol.source.WMTS(wmts_options);
    wmts_layer.setSource(wmts_source);

    return wmts_layer;
};



function project(coordinate) {
    var bdmc = BDMC.convertLL2MC(new ce(coordinate[0], coordinate[1]));
    return [bdmc.lng, bdmc.lat];
}

function unproject(coordinate) {
    var bd09 = BDMC.convertMC2LL(new ce(coordinate[0], coordinate[1]));
    return [bd09.lng, bd09.lat];
}
