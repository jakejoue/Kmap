goog.provide('KMap.Graphic');

/**
 * @constructor
 * @api
 */
KMap.Graphic = function (id, options) {
    /**
     * @protected
     */
    this.feature_ = new ol.Feature({
        id: id,
        geometry: undefined,
        attrs: options.attrs
    });
    /**
     * @protected
     */
    this.style_ = undefined;
};

/**
 * @api
 */
KMap.Graphic.prototype.getId = function () {
    return this.feature_.get('id');
};

/**
 * @api
 */
KMap.Graphic.prototype.setId = function (id) {
    return this.feature_.set('id', id);
};

/**
 * @api 
 */
KMap.Graphic.prototype.getGeometry = function () {
    return this.feature_.getGeometry();
};

/**
 * @api
 */
KMap.Graphic.prototype.setGeometry = function (geometry) {
    this.feature_.setGeometry(geometry);
};

/**
 * @api
 */
KMap.Graphic.prototype.getStyle = function () {
    return this.style_;
};

/**
 * @api
 */
KMap.Graphic.prototype.setStyle = function (style) {
    this.style_ = style;
    this.feature_.setStyle(style.getStyle());
};

/**
 * @api
 */
KMap.Graphic.prototype.getAttributes = function () {
    return this.feature_.get('attrs');
};

/**
 * @api
 */
KMap.Graphic.prototype.setAttributes = function (attrs) {
    this.feature_.set('attrs', attrs);
};

/**
 * @api
 */
KMap.Graphic.prototype.getFeature = function () {
    return this.feature_;
};

/**
 * @api
 */
KMap.Graphic.prototype.setFeature = function (feature) {
    this.feature_ = feature;
};
