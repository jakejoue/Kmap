goog.provide('KMap.Symbol');

goog.require('ol.style.Style');

/**
 * @api
 * @constructor
 */
KMap.Symbol = function (options) {
    /**
     * @protected
     */
    this.style_ = new ol.style.Style();
};

KMap.Symbol.prototype.getStyle = function () {
    return this.style_;
};

KMap.Symbol.prototype.getImage = function () {
    return this.style_.getImage();
};

KMap.Symbol.prototype.setImage = function (image) {
    this.style_.setImage(image);
};

KMap.Symbol.prototype.getStroke = function () {
    return this.style_.getStroke();
};

KMap.Symbol.prototype.setStroke = function (stroke) {
    this.style_.setStroke(stroke);
};

KMap.Symbol.prototype.getFill = function () {
    return this.style_.getFill();
};

KMap.Symbol.prototype.setFill = function (fill) {
    this.style_.setFill(fill);
};

KMap.Symbol.prototype.getText = function () {
    return this.style_.getText();
};

KMap.Symbol.prototype.setText = function (text) {
    this.style_.setText(text);
};
