goog.provide('KMap.TextSymbol');

goog.require('KMap.Symbol');
goog.require('ol.style.Fill');
goog.require('ol.style.Stroke');
goog.require('ol.style.Text');

/**
 * @api
 * @constructor
 * @extends {KMap.Symbol}
 */
KMap.TextSymbol = function (options) {
    KMap.Symbol.call(this, options);
    /**
     * @type {ol.style.Text}
     * @private
     */
    this.text_ = new ol.style.Text({
        font: options.font,
        offsetX: options.offsetX,
        offsetY: options.offsetY,
    });
    if (options.stroke) {
        var stroke = new ol.style.Stroke({
            color: options.stroke,
            width: options.width
        });
        this.text_.setStroke(stroke);
    }
    if (options.fill) {
        var fill = new ol.style.Fill({
            color: options.fill
        });
        this.text_.setFill(fill);
    }
    this.setText(this.text_);
};
ol.inherits(KMap.TextSymbol, KMap.Symbol);
