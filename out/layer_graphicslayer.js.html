﻿goog.provide('KMap.GraphicsLayer');

goog.require('KMap');
goog.require('KMap.Layer');

/**
 * @api
 * @constructor
 * @extends {KMap.Layer}
 */
KMap.GraphicsLayer = function (id, options) {
    var baseOptions = ol.obj.assign({}, options);
    this.source_ = undefined;
    KMap.Layer.call(this, id, baseOptions);
};
ol.inherits(KMap.GraphicsLayer, KMap.Layer);

KMap.GraphicsLayer.prototype.createLayer = function (options) {
    this.source_ = new ol.source.Vector();
    var vector_layer = new ol.layer.Vector({
        source: this.source_,
        style: function (feature) {
            return feature.getStyle();
        }
    });
    return vector_layer;
};

/**
 * @api
 */
KMap.GraphicsLayer.prototype.getExtent = function () {
    var layer = this.getLayer();
    return layer.getExtent() || this.source_.getExtent();
};

/**
 * @api
 */
KMap.GraphicsLayer.prototype.addGraphic = function (graphic) {
    this.source_.addFeature(graphic.getFeature());
};

/**
 * @api
 */
KMap.GraphicsLayer.prototype.removeGraphic = function (graphic) {
    this.source_.removeFeature(graphic.getFeature());
};

/**
 * @api
 */
KMap.GraphicsLayer.prototype.clearGraphic = function () {
    this.source_.clear();
};

/**
 * @api
 */
KMap.GraphicsLayer.prototype.addGraphics = function (graphics) {
    var features = [];
    for(var i=0;i<graphics.length;i++) {
        features.push(graphics[i].getFeature());
    }
    this.source_.addFeatures(features);
};
